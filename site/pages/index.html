Get up and running in seconds.

Quick-start Instructions

# Download tackler-cli-{{ site.version }}.jar or build it.

~ $ mkdir -p tackler/txns; cd tackler

~/tackler $ cp /path/to/tackler-cli-{{ site.version }}.jar .

~/tackler $ cat > txns/journal.txn << EOF 

2018-05-01 'It was warm, sunny day
  Expenses:Ice_cream  2
  Assets:Cash
EOF

~/tackler $ java -jar ./tackler-cli-{{ site.version }}.jar --accounts.strict=false 

BALANCE
-------
                 0.00   -2.00  Assets
                -2.00   -2.00  Assets:Cash
                 0.00    2.00  Expenses
                 2.00    2.00  Expenses:Ice_cream
=====================
                 0.00
